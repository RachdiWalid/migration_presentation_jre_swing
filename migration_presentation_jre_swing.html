class _State extends State<BlocReorderPage> {
  var blocs = [
    {'id': '1', 'titre': 'Accroche'},
    {'id': '2', 'titre': 'Corps email'},
    {'id': '3', 'titre': 'CTA'},
    {'id': '4', 'titre': 'Méta SEO'},
  ];

  @override
  Widget build(BuildContext context) {
    return ReorderableListView.builder(
      // proxyDecorator = widget fantôme pendant drag
      proxyDecorator: (child, index, anim) =>
        AnimatedBuilder(
          animation: anim, child: child,
          builder: (ctx, c) => Material(
            elevation: Tween(0.0, 8.0)
              .animate(anim).value,
            child: c,
          ),
        ),
      itemCount: blocs.length,
      onReorder: (oldIdx, newIdx) {
        // Correction obligatoire si on descend
        if (newIdx > oldIdx) newIdx--;
        setState(() {
          final item = blocs.removeAt(oldIdx);
          blocs.insert(newIdx, item);
        });
      },
      itemBuilder: (ctx, i) => ListTile(
        key: ValueKey(blocs[i]['id']), // requis!
        leading: const Icon(Icons.drag_handle),
        title: Text(blocs[i]['titre']!),
      ),
    );
  }
}

// Package: super_drag_and_drop + file_picker
import 'package:super_drag_and_drop/...';
import 'package:file_picker/file_picker.dart';

DropRegion(
  formats: Formats.standardFormats,
  onDropOver: (event) {
    setState(() => _hovering = true);
    return DropOperation.copy;
  },
  onDropLeave: (_) =>
    setState(() => _hovering = false),
  onPerformDrop: (event) async {
    setState(() => _hovering = false);
    final files = <XFile>[];
    for (final item in event.session.items) {
      await item.dataReader!.getValue(
        Formats.fileUri,
        (uri) { if (uri != null)
          files.add(XFile(uri.path)); });
    }
    widget.onFilesAdded(files);
  },
  child: GestureDetector(
    // Fallback dialog natif
    onTap: () async {
      final r = await FilePicker.platform
        .pickFiles(allowMultiple: true);
      if (r != null)
        widget.onFilesAdded(r.xFiles);
    },
    child: DropZoneWidget(hovering: _hovering),
  ),
)

Code Dart — brief_form.dart
brief_form.dart
class _BriefFormState
    extends State<BriefForm> {
  final _formKey = GlobalKey<FormState>();
  final _titreCtrl  = TextEditingController();
  final _clientCtrl = TextEditingController();
  final _descCtrl   = TextEditingController();
  String? _ton;

  @override
  void dispose() { // libérer les controllers
    _titreCtrl.dispose();
    _clientCtrl.dispose();
    _descCtrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) =>
    Form(
      key: _formKey,
      // Validation au fil de la saisie
      autovalidateMode:
        AutovalidateMode.onUserInteraction,
      child: Column(children: [

        TextFormField(
          controller: _titreCtrl,
          decoration: const InputDecoration(
            labelText: 'Titre du projet'),
          validator: (v) {
            if (v?.isEmpty ?? true)
              return 'Champ obligatoire';
            if (v!.length < 4)
              return 'Min. 4 caractères';
            return null; // valide
          },
        ),

        DropdownButtonFormField<String>(
          value: _ton,
          decoration: const InputDecoration(
            labelText: 'Ton éditorial'),
          items: ['Professionnel','Créatif']
            .map((t) => DropdownMenuItem(
              value: t, child: Text(t)))
            .toList(),
          validator: (v) =>
            v == null ? 'Requis' : null,
          onChanged: (v) =>
            setState(() => _ton = v),
        ),

        ElevatedButton(
          onPressed: () {
            if (_formKey.currentState!.validate()) {
              _formKey.currentState!.save();
            }
          },
          child: const Text('Créer le brief'),
        ),
      ]),
    );
}
